from collections import defaultdict

# Sample input
text = """MapReduce is simple and powerful. MapReduce counts words easily."""

# Map step
mapped = [(word.lower(), 1) for word in text.split()]

# Reduce step
word_count = defaultdict(int)
for word, count in mapped:
    word_count[word.strip('.,!')] += count

# Output
for word, count in word_count.items():
    print(word, count)
